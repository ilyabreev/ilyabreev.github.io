packages 

package main

import (
    "fmt"
    "math/rand"
    "time"
)

func main() {
    time := time.Now().UnixNano()
    fmt.Println("Current time is", time)
    rand.Seed(time)
    fmt.Println("My favorite number is", rand.Intn(10))
}
functions 

package main

import "fmt"

func swap(x, y int) (int, int) {
    return y, x
}

func main() {
    a, b := swap(1, 2)
    fmt.Println(a, b)
}

package main

import "fmt"

func main() {
    fmt.Println("counting")
    for i := 0; i < 10; i++ {
        defer fmt.Println(i)
    }

}
